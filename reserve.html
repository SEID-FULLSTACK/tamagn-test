Provide helpful info about real estate and bank loans in Amharic and English."; const result = await model.generateContent(`${context} \n User: ${message}`); const response = await result.response; res.json({ reply: response.text() }); } catch (error) { console.error("=== እውነተኛው የስህተት ዝርዝር (Error Details) ==="); console.error(error); // ይህ ስህተቱን በሙሉ በተርሚናል ላይ ያሳያል res.status(500).json({ error: "AI Server Error", details: error.message // ስህተቱን ለብሮውዘርህም ይልካል }); } }); // ይህን ኮድ በ server.js ውስጥ አስገባ app.get('/list-models', async (req, res) => { try { const models = await genAI.listModels(); res.json(models.models); } catch (error) { res.status(500).json({ error: error.message }); } }); const PORT = process.env.PORT || 5000; app.listen(PORT, () => console.log(`Server started on port ${PORT}`));
